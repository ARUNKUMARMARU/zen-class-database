1.Find all the topics and tasks which are taught in the month of October

 Query : db.tasks.aggregate([
  {
    '$match': {
      'taught month': 'october'
    }
  }, {
    '$lookup': {
      'from': 'topics', 
      'localField': 'taught month', 
      'foreignField': 'taught month', 
      'as': 'results'
    }
  }
])
